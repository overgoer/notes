# Java — Прогресс обучения

```dataviewjs
// =============================================
// Прогресс обучения Java — автоматический
// =============================================

const javaTasks = dv.pages('"eddytester/Java/Бэклог задач.md"')
    .file.tasks
    .filter(t => t.tags.includes("#java") && !t.text.toLowerCase().includes("daily"));

const total = javaTasks.length;
const completed = javaTasks.filter(t => t.completed).length;
const pct = total === 0 ? 0 : Math.round(completed / total * 100);

const phases = [
    { name: "Фаза 0: Фундамент",     weeks: [1,2,3,4],   color: "#4ecdc4" },
    { name: "Фаза 1: Core для собеса", weeks: [5,6,7,8], color: "#45b7d1" },
    { name: "Фаза 2: Инструменты",   weeks: [9,10,11],   color: "#96ceb4" },
    { name: "Фаза 3: Алгоритмы лайт", weeks: [12,13],    color: "#ffeaa7" },
    { name: "Фаза 4: System Design", weeks: [14,15],     color: "#f39c12" },
    { name: "Фаза 5: Mock-сессии",   weeks: [16,17,18],  color: "#e17055" },
];

function progressBar(pct, color) {
    const filled = Math.round(pct / 5);
    const empty = 20 - filled;
    return `<span style="font-family:monospace;font-size:14px; white-space:nowrap">
        <span style="color:${color}">${'█'.repeat(filled)}</span>
        <span style="color:#2d2d2d">${'█'.repeat(empty)}</span>
        <span style="margin-left:8px;color:#aaa">${pct}%</span>
    </span>`;
}

// === ОБЩИЙ ПРОГРЕСС ===
dv.paragraph(`## 🎯 Общий прогресс: ${completed}/${total} задач — ~сен 2026`);
dv.paragraph(progressBar(pct, "#4ecdc4"));

// === ПРОГРЕСС ПО ФАЗАМ ===
for (const phase of phases) {
    const phaseTasks = javaTasks.filter(t =>
        phase.weeks.some(w => t.text.includes(`Неделя ${w}`))
    );
    const phaseTotal = phaseTasks.length;
    const phaseDone = phaseTasks.filter(t => t.completed).length;
    const phasePct = phaseTotal === 0 ? 0 : Math.round(phaseDone / phaseTotal * 100);

    dv.paragraph(`**${phase.name}** — ${phaseDone}/${phaseTotal}`);
    dv.paragraph(progressBar(phasePct, phase.color));
}

// === СТАТИСТИКА ===
dv.paragraph("---");
dv.paragraph("### 📊 Статистика");

const weekTasks = javaTasks.filter(t => t.text.includes("Неделя"));
const weeksDone = weekTasks.filter(t => t.completed).length;
const weeksTotal = weekTasks.length;

dv.paragraph([
    `✅ Завершено недель: ${weeksDone}/${weeksTotal}`,
    `📅 Всего задач: ${completed}/${total}`,
    `🎯 Осталось: ${total - completed}`,
].join("  \n"));

// === ПОСЛЕДНИЕ ВЫПОЛНЕННЫЕ ===
dv.paragraph("---");
dv.paragraph("### ✅ Последние выполненные");
const recent = javaTasks
    .filter(t => t.completed)
    .sort(t => t.text)
    .slice(-3);
if (recent.length > 0) {
    for (const t of recent) {
        dv.paragraph(`- ~~${t.text}~~`);
    }
} else {
    dv.paragraph("Пока нет завершённых задач. Открой Неделю 1!");
}
```
